// ═══════════════════════════════════════════════════
//  รายงานการเข้าออก — Google Apps Script
//  Sheet: รายงานบันทึกการเข้าออก
// ═══════════════════════════════════════════════════
const SHEET_NAME = 'รายงานบันทึกการเข้าออก';

function doGet(e) {
  const p = e.parameter;
  const action = p.action || '';
  if (action === 'ping')    return ok({ pong: true });
  if (action === 'getData') return ok({ rows: getData(p) });
  if (action === 'getMeta') return ok(getMeta());
  return ok({ status: 'ready' });
}

function getData(p) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName(SHEET_NAME);
  if (!sh) return [];
  const data = sh.getDataRange().getValues();
  // หาแถว header (แถวที่มี "วันที่" เป็น cell แรกที่ไม่ว่าง)
  let hdrRow = -1;
  for (let i = 0; i < Math.min(10, data.length); i++) {
    const row = data[i].map(c => String(c).trim());
    if (row.some(c => c === 'วันที่' || c.includes('วันที่บันทึก') || c.includes('ทะเบียน'))) {
      hdrRow = i; break;
    }
  }
  if (hdrRow < 0) {
    // fallback: หา row ที่มี cell ไม่ว่างมากที่สุดใน 5 แถวแรก
    let maxFilled = 0;
    for (let i = 0; i < Math.min(5, data.length); i++) {
      const filled = data[i].filter(c => c !== '' && c !== null).length;
      if (filled > maxFilled) { maxFilled = filled; hdrRow = i; }
    }
  }
  const hdrs = data[hdrRow].map(h => String(h).trim());

  const rows = data.slice(hdrRow + 1).map(r => {
    const o = {};
    hdrs.forEach((h, i) => {
      if (!h) return; // ข้าม header ว่าง
      const v = r[i];
      if (v instanceof Date) {
        // ตรวจว่าเป็น time-only (base date 1899-12-30) หรือ datetime จริง
        const epoch = v.getTime();
        const base1899 = new Date('1899-12-30T00:00:00Z').getTime();
        const diffDays = (epoch - base1899) / 86400000;
        if (diffDays < 1) {
          // time-only serial
          o[h] = Utilities.formatDate(v, 'Asia/Bangkok', 'HH:mm:ss');
        } else {
          o[h] = Utilities.formatDate(v, 'Asia/Bangkok', 'dd/MM/yyyy HH:mm:ss');
        }
      } else {
        o[h] = v == null ? '' : String(v).trim();
      }
    });
    return o;
  }).filter(r => { const v = r['วันที่']||''; return v && v !== '' && v !== '—'; });

  // filter ตาม month และ objective ถ้าส่งมา
  let filtered = rows;
  if (p.month) {
    filtered = filtered.filter(r => {
      const ym = getYearMonth(r['วันที่']);
      return ym === p.month;
    });
  }
  if (p.obj && p.obj !== 'ALL') {
    const objQ = String(p.obj).trim().toLowerCase();
    filtered = filtered.filter(r => {
      const val = String(r['วัตถุประสงค์'] || r['วัตถุ'] || '').trim().toLowerCase();
      return val === objQ || val.includes(objQ);
    });
  }

  // เรียงตามวันเวลาเริ่มบันทึก น้อย→มาก
  filtered.sort((a, b) => {
    const ta = parseDateTime(a['วันเวลาเริ่มบันทึก'] || a['วันที่']);
    const tb = parseDateTime(b['วันเวลาเริ่มบันทึก'] || b['วันที่']);
    return ta - tb;
  });

  // limit จำนวน rows ที่ส่งกลับ (ป้องกัน payload ใหญ่เกิน)
  const limit = parseInt(p.limit) || 500;
  return filtered.slice(0, limit);
}

function getMeta() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName(SHEET_NAME);
  if (!sh) return { months: [], objectives: [] };
  const data = sh.getDataRange().getValues();
  let hdrRow = 2;
  for (let i = 0; i < Math.min(10, data.length); i++) {
    if (String(data[i][0]).includes('วันที่')) { hdrRow = i; break; }
  }
  const hdrs = data[hdrRow].map(h => String(h).trim());
  const ymIdx = hdrs.findIndex(h => h.includes('วันที่'));
  const objIdx = hdrs.findIndex(h => h.includes('วัตถุ'));

  const months = new Set();
  const objectives = new Set();
  data.slice(hdrRow + 1).forEach(r => {
    if (r[ymIdx]) {
      const v = r[ymIdx];
      let ym = '';
      if (v instanceof Date) {
        ym = Utilities.formatDate(v, 'Asia/Bangkok', 'yyyy-MM');
      } else {
        // parse dd/MM/yyyy manually
        const s = String(v).trim();
        const parts = s.split('/');
        if (parts.length >= 3) {
          const dd = parts[0], mm = parts[1].padStart(2,'0'), yyyy = parts[2].split(' ')[0];
          ym = yyyy + '-' + mm;
        }
      }
      if (ym) months.add(ym);
    }
    if (r[objIdx] && String(r[objIdx]).trim()) objectives.add(String(r[objIdx]).trim());
  });

  return {
    months: [...months].sort().reverse(),
    objectives: [...objectives].sort()
  };
}

function getYearMonth(dateStr) {
  if (!dateStr) return '';
  if (dateStr instanceof Date) {
    return Utilities.formatDate(dateStr, 'Asia/Bangkok', 'yyyy-MM');
  }
  const parts = String(dateStr).split('/');
  if (parts.length >= 3) {
    // dd/MM/yyyy — parts[0]=dd, parts[1]=MM, parts[2]=yyyy
    const yPart = parts[2].split(' ')[0];
    const m = parts[1].padStart(2, '0');
    return yPart + '-' + m;
  }
  return '';
}

function parseDateTime(s) {
  if (!s) return 0;
  const parts = String(s).split(' ');
  const dp = parts[0].split('/');
  if (dp.length < 3) return 0;
  const dateStr = dp[2].split(' ')[0] + '-' + dp[1].padStart(2,'0') + '-' + dp[0].padStart(2,'0');
  const timeStr = parts[1] || '00:00:00';
  return new Date(dateStr + 'T' + timeStr).getTime();
}

function ok(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
