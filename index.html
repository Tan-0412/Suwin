// ═══════════════════════════════════════════════════ // รายงานการเข้าออก — Google Apps Script v3 // Sheet: รายชื่อบุคคลเข้าออก // ═══════════════════════════════════════════════════ const SHEET_NAME = 'รายชื่อบุคคลเข้าออก'; const COL = { date: 'วันที่บันทึก', plate: 'ทะเบียนรถ', person: 'บุคคลเข้าออก', type: 'ประเภทเข้าออก', group: 'กลุ่มบุคคลเข้าออก', dept: 'ฝ่าย/ฝ่ายที่ต้องการติดต่อ', obj: 'วัตถุประสงค์', inDT: 'วันเวลาเริ่มบันทึก', inBy: 'ผู้บันทึกเริ่ม', outDT: 'วันเวลาจบบันทึก', outBy: 'ผู้บันทึกจบ', }; function doGet(e) { const p = e.parameter; const action = p.action || ''; if (action === 'ping') return ok({ pong: true }); if (action === 'getData') return ok({ rows: getData(p) }); if (action === 'getMeta') return ok(getMeta()); return ok({ status: 'ready' }); } function toYearMonth(v) { if (!v && v !== 0) return ''; if (v instanceof Date) { return Utilities.formatDate(v, 'Asia/Bangkok', 'yyyy-MM'); } const s = String(v).trim().split(' ')[0]; const parts = s.split('/'); if (parts.length >= 3) { const y = parts[2].substring(0, 4); const m = parts[1].padStart(2, '0'); if (y.length === 4 && !isNaN(parseInt(y))) return y + '-' + m; } return ''; } function findHdrRow(data) { for (let i = 0; i < Math.min(10, data.length); i++) { const row = data[i].map(c => String(c).trim()); if (row.some(c => c === 'ทะเบียนรถ' || c === 'บุคคลเข้าออก' || c === 'วันที่บันทึก')) { return i; } } return 0; } function getData(p) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sh = ss.getSheetByName(SHEET_NAME); if (!sh) return []; const data = sh.getDataRange().getValues(); const hdrRow = findHdrRow(data); const hdrs = data[hdrRow].map(h => String(h).trim()); const rows = data.slice(hdrRow + 1).map(r => { const o = {}; hdrs.forEach((h, i) => { if (!h) return; const v = r[i]; if (v instanceof Date) { const epoch = v.getTime(); const base1899 = new Date('1899-12-30T00:00:00Z').getTime(); const diffDays = (epoch - base1899) / 86400000; if (diffDays < 1) { o[h] = Utilities.formatDate(v, 'Asia/Bangkok', 'HH:mm:ss'); } else { o[h] = Utilities.formatDate(v, 'Asia/Bangkok', 'dd/MM/yyyy HH:mm:ss'); } } else { o[h] = v == null ? '' : String(v).trim(); } }); return o; }).filter(r => { const d = r[COL.date] || ''; const pl = r[COL.plate] || ''; const indt = r[COL.inDT] || ''; return d !== '' || pl !== '' || indt !== ''; }); let filtered = rows; if (p.month) { filtered = filtered.filter(r => { const ym = toYearMonth(r[COL.date] || r[COL.inDT] || ''); return ym === p.month; }); } filtered.sort((a, b) => { const ta = parseDateTime(a[COL.inDT] || a[COL.date] || ''); const tb = parseDateTime(b[COL.inDT] || b[COL.date] || ''); return tb - ta; }); const limit = parseInt(p.limit) || 2000; return filtered.slice(0, limit); } function getMeta() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sh = ss.getSheetByName(SHEET_NAME); if (!sh) return { months: [], objectives: [] }; const data = sh.getDataRange().getValues(); const hdrRow = findHdrRow(data); const hdrs = data[hdrRow].map(h => String(h).trim()); // หา index วันที่ — เอาเฉพาะที่ match ตรงเป๊ะกับ COL.date หรือ COL.inDT เท่านั้น // ไม่ใช้ includes() เพื่อไม่ให้ match column เวลา-เท่านั้น เช่น 'เริ่มบันทึก' const dateIdxList = []; const exactDateIdx = hdrs.findIndex(h => h === COL.date); const exactInDTIdx = hdrs.findIndex(h => h === COL.inDT); if (exactDateIdx >= 0) dateIdxList.push(exactDateIdx); if (exactInDTIdx >= 0) dateIdxList.push(exactInDTIdx); const objIdx = hdrs.findIndex(h => h === COL.obj || h.includes('วัตถุ')); const months = new Set(); const objectives = new Set(); data.slice(hdrRow + 1).forEach(r => { // ลอง parse เดือนจากทุก date column — เจออันแรกที่ได้ผลก็พอ for (const idx of dateIdxList) { const ym = toYearMonth(r[idx]); if (ym) { months.add(ym); break; } } if (objIdx >= 0 && r[objIdx] && String(r[objIdx]).trim()) { objectives.add(String(r[objIdx]).trim()); } }); return { months: [...months].sort().reverse(), objectives: [...objectives].sort() }; } function parseDateTime(s) { if (!s) return 0; const parts = String(s).split(' '); const dp = parts[0].split('/'); if (dp.length < 3) return 0; const y = dp[2].substring(0, 4); const m = dp[1].padStart(2, '0'); const d = dp[0].padStart(2, '0'); const timeStr = parts[1] || '00:00:00'; return new Date(y + '-' + m + '-' + d + 'T' + timeStr).getTime(); } function ok(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
