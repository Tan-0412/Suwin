📄 Apps Script (คัดลอกทั้งหมด) 
// ═══════════════════════════════════════════════════
//  รายงานการเข้าออก — Google Apps Script
//  Sheet: รายชื่อบุคคลเข้าออก
// ═══════════════════════════════════════════════════
const SHEET_NAME = 'รายชื่อบุคคลเข้าออก';

// mapping ชื่อ column ในชีท → ชื่อที่ใช้ใน API
// แก้ตรงนี้ถ้าชื่อ column ใน Sheets เปลี่ยน
const COL = {
  date:    'วันที่บันทึก',
  plate:   'ทะเบียนรถ',
  person:  'บุคคลเข้าออก',
  type:    'ประเภทเข้าออก',
  group:   'กลุ่มบุคคลเข้าออก',
  dept:    'ฝ่าย/ฝ่ายที่ต้องการติดต่อ',
  obj:     'วัตถุประสงค์',
  inDT:    'วันเวลาเริ่มบันทึก',
  inBy:    'ผู้บันทึกเริ่ม',
  outDT:   'วันเวลาจบบันทึก',
  outBy:   'ผู้บันทึกจบ',
};

function doGet(e) {
  const p = e.parameter;
  const action = p.action || '';
  if (action === 'ping')    return ok({ pong: true });
  if (action === 'getData') return ok({ rows: getData(p) });
  if (action === 'getMeta') return ok(getMeta());
  return ok({ status: 'ready' });
}

function getData(p) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName(SHEET_NAME);
  if (!sh) return [];
  const data = sh.getDataRange().getValues();

  // หาแถว header — หา row ที่มี 'ทะเบียนรถ' หรือ 'บุคคลเข้าออก'
  let hdrRow = 0;
  for (let i = 0; i < Math.min(10, data.length); i++) {
    const row = data[i].map(c => String(c).trim());
    if (row.some(c => c === 'ทะเบียนรถ' || c === 'บุคคลเข้าออก' || c === 'เริ่มบันทึก')) {
      hdrRow = i; break;
    }
  }
  const hdrs = data[hdrRow].map(h => String(h).trim());

  const rows = data.slice(hdrRow + 1).map(r => {
    const o = {};
    hdrs.forEach((h, i) => {
      if (!h) return;
      const v = r[i];
      if (v instanceof Date) {
        const epoch = v.getTime();
        const base1899 = new Date('1899-12-30T00:00:00Z').getTime();
        const diffDays = (epoch - base1899) / 86400000;
        if (diffDays < 1) {
          o[h] = Utilities.formatDate(v, 'Asia/Bangkok', 'HH:mm:ss');
        } else {
          o[h] = Utilities.formatDate(v, 'Asia/Bangkok', 'dd/MM/yyyy HH:mm:ss');
        }
      } else {
        o[h] = v == null ? '' : String(v).trim();
      }
    });
    return o;
  }).filter(r => {
    // กรองแถวว่าง — ต้องมีวันที่บันทึก หรือ ทะเบียนรถ
    const d = r[COL.date] || '';
    const pl = r[COL.plate] || '';
    return (d !== '' && d !== '—') || pl !== '';
  });

  // filter ตาม month
  let filtered = rows;
  if (p.month) {
    filtered = filtered.filter(r => {
      const ym = getYearMonth(r[COL.date] || r[COL.inDT] || '');
      return ym === p.month;
    });
  }

  // เรียงตามวันเวลาเริ่มบันทึก น้อย→มาก
  filtered.sort((a, b) => {
    const ta = parseDateTime(a[COL.inDT] || a[COL.date] || '');
    const tb = parseDateTime(b[COL.inDT] || b[COL.date] || '');
    return ta - tb;
  });

  const limit = parseInt(p.limit) || 1000;
  return filtered.slice(0, limit);
}

function getMeta() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sh = ss.getSheetByName(SHEET_NAME);
  if (!sh) return { months: [], objectives: [] };
  const data = sh.getDataRange().getValues();

  let hdrRow = 0;
  for (let i = 0; i < Math.min(10, data.length); i++) {
    const row = data[i].map(c => String(c).trim());
    if (row.some(c => c === 'ทะเบียนรถ' || c === 'บุคคลเข้าออก')) { hdrRow = i; break; }
  }
  const hdrs = data[hdrRow].map(h => String(h).trim());
  const dateIdx = hdrs.findIndex(h => h === COL.date || h === COL.inDT || h.includes('วันที่'));
  const objIdx  = hdrs.findIndex(h => h === COL.obj || h.includes('วัตถุ'));

  const months = new Set();
  const objectives = new Set();
  data.slice(hdrRow + 1).forEach(r => {
    if (r[dateIdx]) {
      const v = r[dateIdx];
      let ym = '';
      if (v instanceof Date) {
        ym = Utilities.formatDate(v, 'Asia/Bangkok', 'yyyy-MM');
      } else {
        const s = String(v).trim().split(' ')[0];
        const parts = s.split('/');
        if (parts.length >= 3) {
          ym = parts[2].padStart(4,'0') + '-' + parts[1].padStart(2,'0');
        }
      }
      if (ym) months.add(ym);
    }
    if (r[objIdx] && String(r[objIdx]).trim()) objectives.add(String(r[objIdx]).trim());
  });

  return {
    months: [...months].sort().reverse(),
    objectives: [...objectives].sort()
  };
}

function getYearMonth(dateStr) {
  if (!dateStr) return '';
  if (dateStr instanceof Date) {
    return Utilities.formatDate(dateStr, 'Asia/Bangkok', 'yyyy-MM');
  }
  const parts = String(dateStr).split('/');
  if (parts.length >= 3) {
    // dd/MM/yyyy — parts[0]=dd, parts[1]=MM, parts[2]=yyyy
    const yPart = parts[2].split(' ')[0];
    const m = parts[1].padStart(2, '0');
    return yPart + '-' + m;
  }
  return '';
}

function parseDateTime(s) {
  if (!s) return 0;
  const parts = String(s).split(' ');
  const dp = parts[0].split('/');
  if (dp.length < 3) return 0;
  const dateStr = dp[2].split(' ')[0] + '-' + dp[1].padStart(2,'0') + '-' + dp[0].padStart(2,'0');
  const timeStr = parts[1] || '00:00:00';
  return new Date(dateStr + 'T' + timeStr).getTime();
}

function ok(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
